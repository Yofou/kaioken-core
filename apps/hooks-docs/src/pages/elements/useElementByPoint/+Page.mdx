# useElementByPoint

Reactive elementByPoint

<UseElementByPointExample />

## Usage

```tsx twoslash
import { useMouse, useElementByPoint } from "@kaioken-core/hooks"
import { Signal } from "kaioken"
const { mouse } = useMouse()
// ---cut---
const { element } = useElementByPoint({
  x: mouse.value.x,
  y: mouse.value.y,
  multiple: false,
  immediate: true,
})

type UseElementByPointReturn<M extends boolean = false> = {
  element: M extends true ? Signal<HTMLElement[]> : Signal<HTMLElement | null>
  start: () => void
  stop: () => void
  isActive: boolean
}
```
